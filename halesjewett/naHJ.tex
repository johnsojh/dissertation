\begin{defn}
  Let $S$ be a free semigroup on the alphabet $A$ and let $v$ be a
  variable which is not a member of $A$.
  \begin{itemize}
    \item[(a)] A \textsl{variable word} $w(v)$ is a member of the free
      semigroup on the alphabet $A \cup \{v\}$ in which $v$ appears. 
  \end{itemize}
\end{defn}
